var op: string;
var op1: int;
var op2: int;
var result: int;
begin
write 'Introduce un entero: ';
read op1;
write 'Introduce otro entero: ';
read op2;
read op;
write 'Introduce un operador (+,-,*,/): ';
read op;
if (op == '+') then
	write op1 + op2; nl;
else 
 if (op == '-') then
	write op1 - op2; nl;
 else
  if (op == '/') then
	write op1 / op2; nl;
  else
   if (op == '*') then
	write op1 * op2; nl;
   else
    write 'Operador desconocido '; nl;
   end
  end 
 end
end
nl; write 'Fichero finalizado! '; 
end.
